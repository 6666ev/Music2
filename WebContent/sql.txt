create table songs
(
	song_id	 serial,
	song_name	 varchar(100),
	song_time	 varchar(100),
	primary key (song_id) 

);

create table singers
(
	singer_id	 	 serial,
	singer_name	 varchar(100),
	singer_gender	 varchar(5),
	singer_nationality	 varchar(30),		

	primary key(singer_id)

);

create table albums
(
	album_id	 serial,
	album_name	 varchar(100),
	singer_id	 int,

	primary key(album_id)

);

create table tags
(
	tag_id	 serial,
	tag_name	 varchar(100),

	primary key(tag_id)
);


create table listened
(
	email 	  varchar(100),
	song_id	  int,
	
	primary key(email, song_id)
	
);

create table liked
(
	email 	 varchar(100),
	song_id	 int,
	
	primary key(email, song_id)
);


create table song_tags
(
	song_id	 int,
	tag_id	 int,

	primary key(song_id,tag_id)
);

create table song_albums
(
	song_id	 int,
	album_id	 int,

	primary key(song_id,album_id)	

);

create table song_singers
(
	song_id	 int,
	singer_id	 int,

	primary key (song_id,singer_id)

);
